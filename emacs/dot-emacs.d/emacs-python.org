* Package Settings
  #+BEGIN_SRC elisp
    (use-package python
      :mode ("\\.py\\'" . python-mode)
      :interpreter ("python3" . python-mode)
      
      :init (setq-default indent-tabs-mode nil)
      
      :config
        (setq python-indent-offset 4)
        (add-hook 'python-mode-hook 'smartparens-mode)
        (add-hook 'python-mode-hook 'color-identifiers-mode))
  #+END_SRC
